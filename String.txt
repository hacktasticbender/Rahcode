import java.util.Scanner;

public class ExtractString 
{
	public static void main(String args[])
	{
		Scanner pt=new Scanner(System.in);
		String str;
		int i,j;
		System.out.println("enter the text");
		str=pt.next();
		System.out.println("enter index values");
		i=pt.nextInt();
		j=pt.nextInt();
		if(i<0|j<0|i>str.length()-1|j>str.length()-1|i>j)
		{
			System.out.println("invalid index");
			System.exit(0);
		}
		char[] newAry=new char[10];
		for(int a=i,b=0;a<=j;a++,b++)
		{
			newAry[b]=str.charAt(a);
		}
		String newstr=new String(newAry);
		System.out.println("substring is"+ newstr);
	}
}
